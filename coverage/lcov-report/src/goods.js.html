function Goods(id, name, price, kind, isImportDuty) {
  // 商品编号 Number
  this.id = id;
 
  // 商品名称 String
  this.name = name;
 
  // 商品价格 String
  this.price = price || '0.00';
 
  // 商品种类 Number 0为食物，1为书籍，2为药品，100为其他
  this.kind = kind;
 
  // 商品是否为进口 Boolean
  this.isImportDuty = isImportDuty;
}
